import pyodbc
import pandas as pd

# Define the connection parameters
server = 'your_server_name'  # e.g., 'localhost\\SQLEXPRESS'
database = 'your_database_name'
username = 'your_username'
password = 'your_password'

# Create the connection string
connection_string = f'DRIVER={{ODBC Driver 17 for SQL Server}};SERVER={server};DATABASE={database};UID={username};PWD={password}'

# Establish the connection
connection = pyodbc.connect(connection_string)

# Create a cursor object using the connection
cursor = connection.cursor()

# SQL queries to retrieve data from the tables
employee_query = 'SELECT * FROM Employees'
childcare_query = 'SELECT * FROM ChildCareCenters'

# Load data into pandas DataFrames
employees_df = pd.read_sql(employee_query, connection)
childcare_df = pd.read_sql(childcare_query, connection)

# Close the connection
cursor.close()
connection.close()

# Display the first few rows of each DataFrame
print("Employees DataFrame:")
print(employees_df.head())

print("\nChild Care Centers DataFrame:")
print(childcare_df.head())
